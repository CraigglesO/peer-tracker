#tag the file
docker tag 4d7dc0ffcee2 oconnorct1/peertracker:latest


# get the logs from a docker image:
docker logs 4d7dc0ffcee2

# get basic running information from a docker image:
docker inspect 4d7dc0ffcee2

#Get redis up and running:
docker run --name redis -d -p 6379:6379 oconnorct1/redis
docker run -d --name peertracker --link redis:redis -p 1337:1337 -p 1337:1337/udp -p 8080:8080 -i -t oconnorct1/peertracker:latest
